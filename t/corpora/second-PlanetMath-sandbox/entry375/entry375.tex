\textbf{Definition:} \newline Let $A=(a_{ij})$ be a square matrix of
order $n$.  The matrix $A$ is \emph{symmetric} if $a_{ij} = a_{ji}$
for all $1 \leq i \leq n, 1 \leq j \leq n$.
\begin{center}$A =
\begin{pmatrix}
  a_{11} & \cdots & a_{1n} \\
  \vdots & \ddots & \vdots \\
  a_{n1} & \cdots & a_{nn}
\end{pmatrix}$
\end{center}

\textbf{Properties:}
\begin{enumerate}
  \item $A^t = A$ where $A^t$ is the matrix transpose
\end{enumerate}

\textbf{Examples:}
\begin{itemize}
  \item $\begin{pmatrix}
    a & b \\
    b & c 
  \end{pmatrix}$
  \item $\begin{pmatrix}
    a & b & c \\
    b & d & e \\
    c & e & f 
  \end{pmatrix}$
\end{itemize}
